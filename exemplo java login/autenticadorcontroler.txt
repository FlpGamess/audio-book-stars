package com.exemplo.demo.controller;

import com.exemplo.demo.model.Usuario;
import com.exemplo.demo.repository.UsuarioRepository;
import com.exemplo.demo.security.JwtUtil;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

import java.util.Optional;

@RestController
@RequestMapping("/api/auth")
public class AuthController {

    @Autowired
    private UsuarioRepository usuarioRepository;

    @Autowired
    private JwtUtil jwtUtil;

    // Login: retorna token JWT
    @PostMapping("/login")
    public ResponseEntity<?> login(@RequestBody Usuario usuario) {
        Optional<Usuario> encontrado = usuarioRepository.findByEmail(usuario.getEmail());

        if (encontrado.isEmpty() || !encontrado.get().getSenha().equals(usuario.getSenha())) {
            return ResponseEntity.badRequest().body("Email ou senha incorretos");
        }

        // gera token
        String token = jwtUtil.generateToken(encontrado.get().getEmail());

        return ResponseEntity.ok(token);
    }

    // Exemplo de endpoint protegido
    @GetMapping("/me")
    public ResponseEntity<?> perfil(@RequestHeader("Authorization") String authHeader) {
        if (authHeader == null || !authHeader.startsWith("Bearer ")) {
            return ResponseEntity.status(401).body("Token ausente");
        }

        String token = authHeader.substring(7); // remove "Bearer "
        if (!jwtUtil.validateToken(token)) {
            return ResponseEntity.status(401).body("Token inv√°lido");
        }

        String email = jwtUtil.getEmailFromToken(token);
        Optional<Usuario> usuario = usuarioRepository.findByEmail(email);
        return ResponseEntity.ok(usuario.orElse(null));
    }
}
